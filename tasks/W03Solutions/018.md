class Solution {
public:
    vector<int> preorderTraversal(TreeNode* root) {
        vector<int> ans;
        stack<TreeNode*> store;
        while(!store.empty() || root){
            if(root){
                ans.push_back(root->val);
                if(root->right){ 
                    store.push(root->right);
                }
                root=root->left;
            }
            else{
                root=store.top();
                store.pop();
            }
        }
		 return ans;
    }
};
