```c++
class Solution {
public:
    string longestCommonPrefix(vector<string>& strs) {
        if (strs.size() == 0) {
        return "";
    }

    string common = "";
    for (int i = 0;; i++) {
        for (auto word : strs) {
            if (i >= word.size() || word[i] != strs[0][i]) {
                return common;
            }
        }
        common += strs[0][i];
    }
    return common;
    }
};
```
