```c++
#include <iostream>
using namespace std;
 
int main() {
    long long a, k, b, m, x;
 
    cin >> a >> k >> b >> m >> x;
 
    long long left = 0;
    long long right = 2 * x / (a + b) + 1;
    long long t;
    long long d1, d2;
    while (left + 1 < right) {
     t = (left + right) / 2;
     d1 = a * (k - 1) * (t / k) + a * (t % k);
     d2 = b * (m - 1) * (t / m) + b * (t % m);
     if (d1 + d2 >= x)
     right = t;
     else
     left = t;
    }
 
    cout << right;
}
```
