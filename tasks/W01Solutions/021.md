class Solution {
public:
    int removeElements(vector<int>& A, int n, int elem) {
	int p = 0;
	int q = n - 1;
	while( p <= q )
	{
		if( A[p] != elem )
			p++;
		else if( A[q] == elem )
			q--;
		else
		{
			int tmp = A[p];
			A[p++] = A[q];
			A[q--] = tmp;
		}
	}

	return p;
}
    int removeElement(vector<int>& nums, int val) {
        int n = size(nums);
        return removeElements(nums,n,val);
    }
};
